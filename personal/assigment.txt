<script>
    const app = {
        tasks: JSON.parse(localStorage.getItem('tasks')) || [],
        xp: parseInt(localStorage.getItem('xp')) || 0,

        init() {
            this.render();
            this.updateXP(0);
        },

        updateXP(amt) {
            this.xp += amt;
            const level = Math.floor(this.xp / 100) + 1;
            const currentXP = this.xp % 100;
            
            document.getElementById('lvl-val').innerText = level;
            document.getElementById('xp-val').innerText = currentXP;
            document.getElementById('xp-fill').style.width = currentXP + '%';
            
            localStorage.setItem('xp', this.xp);
        },

        addTask() {
            const input = document.getElementById('task-in');
            const cat = document.getElementById('cat-in');
            if(!input.value.trim()) return;

            const newTask = {
                id: Date.now(),
                text: input.value,
                category: cat.value,
                completed: false
            };

            this.tasks.push(newTask);
            this.save();
            input.value = '';
            this.render();
        },

        toggleTask(id) {
            const task = this.tasks.find(t => t.id === id);
            if(!task.completed) {
                this.celebrate();
                this.updateXP(25);
            } else {
                this.updateXP(-25);
            }
            task.completed = !task.completed;
            this.save();
            this.render();
        },

        deleteTask(e, id) {
            e.stopPropagation();
            this.tasks = this.tasks.filter(t => t.id !== id);
            this.save();
            this.render();
        },

        save() {
            localStorage.setItem('tasks', JSON.stringify(this.tasks));
        },

        celebrate() {
            for(let i=0; i<40; i++) {
                const p = document.createElement('div');
                p.className = 'particle';
                const colors = ['#8b5cf6', '#ec4899', '#06b6d4', '#22c55e'];
                p.style.background = colors[Math.floor(Math.random()*colors.length)];
                p.style.width = Math.random() * 10 + 5 + 'px';
                p.style.height = p.style.width;
                p.style.left = Math.random() * 100 + 'vw';
                p.style.top = '-10px';
                p.style.borderRadius = '50%';
                
                document.body.appendChild(p);
                
                const anim = p.animate([
                    { transform: 'translateY(0) rotate(0deg)', opacity: 1 },
                    { transform: `translateY(100vh) rotate(${Math.random()*360}deg)`, opacity: 0 }
                ], { duration: 1500 + Math.random()*1000, easing: 'cubic-bezier(0,0,0.2,1)' });
                
                anim.onfinish = () => p.remove();
            }
        },

        render() {
            const cols = ['top', 'today', 'week'];
            cols.forEach(c => document.getElementById(`list-${c}`).innerHTML = '');

            this.tasks.forEach(t => {
                const li = document.createElement('li');
                li.className = `task-item ${t.completed ? 'completed' : ''}`;
                li.onclick = () => this.toggleTask(t.id);
                li.innerHTML = `
                    <div class="checkbox">${t.completed ? '‚úì' : ''}</div>
                    <span class="task-text">${t.text}</span>
                    <button class="del-btn" onclick="app.deleteTask(event, ${t.id})">‚úï</button>
                `;
                document.getElementById(`list-${t.category}`).appendChild(li);
            });
        }
    };
